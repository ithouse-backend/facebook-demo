// agar yolg'iz githubda ishlashda
git status // qanday o'zgarishlar qilindi
git add . // o'zgartirilgan fayllarni barchasini qo'shish
git commit -m 'nimalar o'zgartirildi' // har bir o'zgargan fayllar uchun aniq qilingan ishni yozish kerak
git push origin (main,uktamova,jahongir, hkz..) // bu yerda har birimizni branchimiz yani aynan biz uchun ajratilgan joyga yuborishimiz kerak. 'main' ga yuborishda extiyot bo'ling, eng asosiy branch main bo'ladi


// agar github orqali ko'dlarni ko'chirib olmoqchi bo'lsak
git fetch origin main // bu yerda ko'dlarni ko'chirib bermaydi, lekin githubda nimalar o'zgargan aytib beradi, keyin bir ko'rib chiqamiz, agar hammasi joyida bo'lsa keyin..
git merge origin/main // githubda o'zgarga barcha fayllarni ko'chirib olamiz


git pull origin main // bu 'git fetch origin main' va 'git merge origin/main' ning bittada qilish usuli. bu ham bo'laveradi lekin birinchi ko'rsatilgan usul yaxshiroq. chunki nimalar o'zgargan bilib olib keyin ko'chirib olamiz
